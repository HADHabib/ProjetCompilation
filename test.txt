class Point(xc: Integer, yc: Integer) {
    this.x := xc;
    this.y := yc;
} is {
    var auto x : Integer;
    var auto y : Integer;

    def print() is {
        ("Point {x: " & this.x.toString() & ", y: " & this.y.toString() & "}").println();
    }
}

{
    x : Point; is
    x := new Point(1, 2);
    x.print();
}
